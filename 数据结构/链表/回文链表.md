#### 回文链表 [leetcode 234](https://leetcode-cn.com/problems/palindrome-linked-list/)
请判断一个链表是否为回文链表。
#### 解法：寻找回文是从中间向两端扩展，判断回文是从两端向中间收缩
1. 遍历链表, 时间复杂度和空间复杂度都是O(N)
```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def isPalindrome(self, head: ListNode) -> bool:
        if not head or not head.next:
            return True
        lst = []
        while head:
            lst.append(head.val)
            head =head.next
        # left, right =0, len(lst)-1
        # while left<right:
        #     if lst[left]!=lst[right]:
        #         return False
        #     left = left+1
        #     right = right-1
        # return True
        return lst ==lst[::-1]
```
2. 利用递归的方式反向遍历链表：后续遍历
```python
class Solution:
    def isPalindrome(self, head: ListNode) -> bool:
        left= head
        def reverse(head):
            nonlocal left
            if not head:
                return True
            if not reverse(head.next):
                return False
            if head.val!=left.val:
                return False
            left = left.next
            return True

        return reverse(head)
```
3. 先翻转后半部分链表，再与前半部分比较，时间复杂度O(N),空间复杂度O(1)
```python
class Solution:
    def isPalindrome(self, head: ListNode) -> bool:
        def reverse(head):
            pre =None
            cur = head
            while cur:
                temp = cur.next
                cur.next = pre
                pre = cur
                cur = temp
            return pre

        if not head or not head.next:
            return True
        slow = fast = head
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next
        if not fast: #链表长度为奇数，slow仍要后移一位
            slow = slow.next

        head2 = reverse(slow)
        while head2:
            if head.val != head2.val:
                return False
            head2 = head2.next
            head = head.next
        return True
```
